def caesar_cipher(text: str, shift: int) -> str:
    """Encrypt text using a Caesar cipher."""
    result = []
    for char in text:
        if char.isalpha():
            base = ord('a') if char.islower() else ord('A')
            result.append(chr((ord(char) - base + shift) % 26 + base))

            result.append(char)
    return ''.join(result)
